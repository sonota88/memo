=== スクリプトの置かれているパス（ディレクトリ）を取得

[source,bash]
--------------------------------
_get_project_dir() {
  local real_path="$(readlink --canonicalize "$0")"
  (
    cd "$(dirname "$real_path")"
    pwd
  )
}
--------------------------------
